\AtBeginDocument{
    % title here, so we can manually \maketitle, which adds the "plain" option to the frame
    \title{\varTitle}
    \subtitle{\varSubtitle}
    \author{\varAuthor}
    \institute{\varInstitute}
    \date{\varDate}

    % title
    \maketitle

    % toc

    \mode<beamer> {
    \begin{frame}[noframenumbering,plain]
        \vspace{10pt}
        \begin{beamercolorbox}[sep=8pt,center]{section title}
            \usebeamerfont{section title}
            Table of Contents
            \vspace{3pt}
        \end{beamercolorbox}
        {
            \small
            \tableofcontents[hidesubsections]
        }
    \end{frame}
    }
}

        \usebeamerfont{section title}

        \vspace{3pt}
    \end{beamercolorbox}

    % tight toc
    {
        \small
        \renewcommand{\baselinestretch}{0.8}

        \renewcommand{\baselinestretch}{1.0}
    }
}
